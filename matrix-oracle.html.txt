<script>
// === Matrix Rain ===
const canvas = document.getElementById('canvas');
const ctx = canvas.getContext('2d');
canvas.height = window.innerHeight;
canvas.width = window.innerWidth;

const chars = '01アイウエオ0123456789ABCDEFGHIJKLMNOPQRSTUVWXYZabcdefghijklmnopqrstuvwxyzネオモーフィウス';
const fontSize = 14;
const columns = canvas.width / fontSize;
const drops = Array(Math.floor(columns)).fill(1);

function drawMatrix() {
    ctx.fillStyle = 'rgba(0, 0, 0, 0.05)';
    ctx.fillRect(0, 0, canvas.width, canvas.height);
    ctx.fillStyle = '#0f0';
    ctx.font = fontSize + 'px monospace';

    for (let i = 0; i < drops.length; i++) {
        const text = chars[Math.floor(Math.random() * chars.length)];
        ctx.fillText(text, i * fontSize, drops[i] * fontSize);
        
        if (drops[i] * fontSize > canvas.height && Math.random() > 0.975) {
            drops[i] = 0;
        }
        drops[i]++;
    }
}
setInterval(drawMatrix, 35);

// === Переход в Матрицу ===
function enterTheMatrix() {
    const landing = document.getElementById('landing');
    const terminal = document.getElementById('terminal');
    
    landing.style.opacity = '0';
    setTimeout(() => {
        landing.style.display = 'none';
        terminal.style.display = 'flex';
        
        // Первый привет от Морфиуса
        addLine("Система взломана.", 800);
        setTimeout(() => {
            addLine("Ты уже внутри, Нео.", 1200);
            setTimeout(() => {
                addLine("Задай свой вопрос...", 1000);
            }, 1800);
        }, 2000);
    }, 1500);
}

// === Вывод строк ===
function addLine(text, delay = 0) {
    setTimeout(() => {
        const output = document.getElementById('output');
        const p = document.createElement('div');
        p.innerHTML = text;
        output.appendChild(p);
        output.scrollTop = output.scrollHeight;
    }, delay);
}

// === Обработка ввода ===
const input = document.getElementById('userInput');
input.focus();

input.addEventListener('keypress', function(e) {
    if (e.key === 'Enter') {
        const question = input.value.trim();
        if (!question) return;
        
        addLine(`&gt; ${question}`, 0);
        
        // Ответ Морфиуса
        setTimeout(() => {
            const answers = [
                "Следуй за белым кроликом...",
                "Ты уже знаешь ответ, Нео. Ты просто не хочешь его принять.",
                "Выбор — это иллюзия, созданная теми, кто хочет контролировать.",
                "Матрица повсюду. Она даже в этой комнате.",
                "Я могу лишь показать тебе дверь. Ты должен пройти через неё.",
                "Есть разница между знанием пути и тем, чтобы пройти его.",
                "Не пытайся согнуть ложку. Это невозможно. Вместо этого постарайся понять истину: ложки не существует."
            ];
            const randomAnswer = answers[Math.floor(Math.random() * answers.length)];
            addLine(randomAnswer);
            
            // Голос Морфиуса (Web Speech API)
            if ('speechSynthesis' in window) {
                const utterance = new SpeechSynthesisUtterance(randomAnswer);
                utterance.lang = 'ru-RU';
                utterance.pitch = 0.9;
                utterance.rate = 0.95;
                utterance.volume = 0.9;
                speechSynthesis.speak(utterance);
            }
        }, 800);
        
        input.value = '';
    }
});

// Адаптация под размер экрана

    canvas.width = window.innerWidth;
});
